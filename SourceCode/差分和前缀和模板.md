# 前缀和

## 一维前缀和
```cpp
s[i] = s[i - 1] + a[i]
a[i] += a[i -1]
```
### 一维前缀和模板题(求出数列某一段中元素的和)
```cpp
#include <iostream>
#include <cstdio>

using namespace std;

const int N = 100010;
int n, m;
int a[N]; //原数组
//int s[N]; //前缀和数组

/*
也可以不用另开一个前缀和数组，直接将前n个数字的和覆盖原数组的值
*/
int main ()
{
    cin >> n >> m;
    for (int i = 1; i <= n; i++)
    {
        scanf("%d", &a[i]);     //读入数据
        //s[i] = s[i - 1] + a[i]; //前缀和数组，请前n个数字的和放到一个前缀和数组中
        a[i] += a[i - 1];
    }
    
    while (m--)                 //循环m次
    {
        int l, r;
        scanf("%d%d", &l, &r);
        //cout << s[r] - s[l - 1] << endl;//将前r个数字-前l - 1个数字，就得到l~r中间的数字
        cout << a[r] - a[l - 1] << endl;
    }
 
    return 0;
}
```

## 二维前缀和

```cpp
//前缀和s[i][k]:
s[i][j] = s[i - 1][j] + s[i][j - 1] - s[i - 1][j - 1] + a[i][j];
//某个子矩阵中元素的和
sum = s[x2][y2] - s[x2][y1 - 1] - s[x1 - 1][y2] + s[x1 - 1][y1 - 1];
```
### 求出子矩阵的和(二位前缀和模板题)
```cpp
#include <iostream>
#include <cstdio>

using namespace std;

int n, m;
int q;
int a[1010][1010]; //原数组
int s[1010][1010]; //前缀和数组

int main ()
{
    cin >> n >> m >> q;
    for (int i = 1; i <= n; i ++)
    {
        for (int j = 1; j <= m; j++)
        {
            scanf("%d", &a[i][j]);
            s[i][j] = s[i][j - 1] + s[i - 1][j] - s[i - 1][j - 1] + a[i][j];//前缀合矩阵,第(i,j)个数到底(1，1)中间的数字之和
        }
    }
    
    while (q--)
    {
        int x1, x2, y1, y2;
        scanf("%d%d%d%d", &x1, &y1, &x2, &y2);
        printf("%d\n", s[x2][y2] - s[x2][y1 - 1] - s[x1 - 1][y2] + s[x1 - 1][y1 - 1]);//子矩阵的和,(x1,y1)到(x2,y2)之间数字的和
    }
    
    return 0;
}
```

## 一维差分模板
差分是用来是数列的一段的数同时+C(常数C)
```cpp
//a[L ~ R] + C:
void insert(int l, int r, int c)
{
	b[l] += c;
	b[r + 1] -= c;
}
```
这样的结果是
1.a[1 ~ L - 1] 无影响
2.a[L ~ R] 每次经过b[L]的时候都会+C,所以前缀和数组中的每个数+C
3.a[R + 1, N]每次经过b[L]+C,经过b[R + 1]-C所以抵消掉了

==b数组的前缀和就是a数组,所以b数组就是在a[i ~i]之间插入了a[i].这样就得到了差分数组==

### 差分模板题
```cpp
#include <iostream>
#include <cstdio>

using namespace std;

const int N = 100010;
int n, m;
int a[N], b[N];//原数组和差分数组

void insert(int l, int r, int c)//差分的核心
{
    b[l] += c;
    b[r + 1] -= c;
}

int main ()
{
    cin >> n >> m;
    for (int i = 1; i <= n; i ++)
    {
        scanf("%d", &a[i]);
        insert(i, i, a[i]);//创建差分数组
    }
    
    while (m --)
    {
        int l, r, c;
        cin >> l >> r >> c;//被加数列的左端点和右端点
        insert(l, r, c);
    }
    
    for (int i = 1; i <= n; i ++)
        a[i] = a[i - 1] + b[i];//求出b数组的前缀和，放到a数组中
        
    for (int i = 1; i <= n; i ++)
        cout << a[i] << ' ';
        
    return 0;
}
```

### 差分矩阵
初始化b差分数组是,可以认为是在原数组上差分一个单位为1的a[i]的元素
差分数组的意义就是:该数组的前缀和数组可以加上某一个常数C

```cpp
#include <iostream>
#include <cstdio>

using namespace std;

const int N = 1010;
int a[N][N], b[N][N];
int n, m, q;

void insert(int x1, int y1, int x2, int y2, int c)
{
    b[x1][y1] += c;
    b[x2 + 1][y1] -= c;
    b[x1][y2 + 1] -= c;
    b[x2 + 1][y2 + 1] += c;
}

int main ()
{
    cin >> n >> m >> q;
    for (int i = 1; i <= n; i ++)
        for (int j = 1; j<= m; j ++)
        {
            scanf("%d", &a[i][j]);
            insert(i, j, i, j, a[i][j]);//使得b成为差分数组
        }
    
    while (q --)
    {
        int x1, y1, x2, y2, c;
        cin >> x1 >> y1 >> x2 >> y2 >> c;
        insert(x1, y1, x2, y2, c);//让某一个子矩阵中所有的元素+c
    }
    
    for (int i = 1; i <= n; i ++)
    {
        for (int j = 1; j <= m; j ++)
        {
            a[i][j] = a[i - 1][j] + a[i][j - 1] - a[i - 1][j - 1] + b[i][j];//求一遍差分数组的前缀和数组
            cout << a[i][j] << ' ';
        }
        cout << endl;
    }
    
    return 0;
}
```
