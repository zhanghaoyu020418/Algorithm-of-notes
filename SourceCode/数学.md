## 质数

### 判断质数(试除法)

```cpp
#include <iostream>

using namespace std;

int n;

bool is_prime(int n)
{
    if(n < 2)
        return false;
    //因为一个数的质因子一定会出现在sqrt(n)之前，所以只要循环到sqrt(n)
    for (int i = 2; i <= n / i; i ++)//sqrt(n)函数太慢，所以可以用i <= n / i
    {
        if(n % i == 0)
            return false;
    }
    return true;
}

int main ()
{
    cin >> n;
    while (n --)
    {
        int num;
        cin >> num;
        if(is_prime(num))//判断素数
            cout << "Yes" << endl;
        else
            cout << "No" << endl;
    }
    return 0;
}
```

### 分解质因数(试除法)
```cpp
#include <iostream>
#include <algorithm>

using namespace std;

int n;

int main ()
{
    cin >> n;
    while (n --)
    {
        int num;
        cin >> num;//输入要分解的数字
        
        //从2循环到num的根号，因为一个数至多只有一个大于sqrt(num)的质因子
        for (int i = 2; i <= num / i; i ++)
        {
            if(num % i == 0)//如果num能整除i说明num是合数
            {
                int cnt = 0;
                while (num % i == 0)//分解合数，直到i不是num的质因子
                {
                    num /= i;
                    cnt ++;
                }
                cout << i << ' ' << cnt << endl;//输出质因子i和对应的质因子个数
            }
        }
        if(num > 1)//如果num！= 1，说明有一个大于sqrt(num)的因子，可以直接输出num
            cout << num << ' ' << 1 << endl;//
        cout << endl;
    }
    return 0;
}
```

### 筛质数(暴力)(埃氏筛法)(线性筛法)

#### 暴力筛法
```cpp
#include <iostream>

using namespace std;

const int N = 1e6 + 10;
int n;
bool st[N];
int cnt;
int prime[N];

int main ()
{
    cin >> n;
    for (int i = 2; i <= n; i ++)
    {
        if(!st[i])//如果没有被筛掉，说明该数字是质数
        {
            //prime[cnt ++] = i;//也可以用一个数组将质数存放到数组中，但是本题没有必要
            cnt ++;
        }
            
        for (int j =  i + i; j <= n; j += i)//i的倍数都是合数，将i的倍数都筛掉，所以从i + i开始筛
            st[j] = true;
    }
    cout << cnt << endl;
    return 0;
}
```

#### 线性筛法(最常用的筛质数的方法)

```cpp
#include <iostream>

using namespace std;

const int N = 10000010;
int n;
bool st[N];
int cnt;
int prime[N];

int main ()
{
    cin >> n;
    for (int i = 2; i <= n; i ++)
    {
        if(!st[i])//如果没有被筛掉，说明该数字是质数
            prime[cnt ++] = i;//将筛出的质数放到一个数组中
        for (int j = 0; prime[j] <= n / i; j ++)//i * prime[j] <= n,i是n的最大质因子
        {
            st[prime[j] * i] = true;//每次都用最小质因数筛掉所有的数
            if(i % prime[j] == 0)
                break;
        }
    }
    cout << cnt << endl;
    return 0;
}
```

## 约数

### 求约数(试除法)

```cpp
#include <iostream>
#include <algorithm>
#include <vector>

using namespace std;

int n;

vector<int> get_divisor(int n)
{
    vector<int> res;
    //因为一个数的约数是成对出现的，所以只要枚举一半就可以了
    for (int i = 1; i <= n / i; i ++)
        if(n % i == 0)
        {
            res.push_back(i);
            if(n / i != i)
                res.push_back(n / i);//如果约数不相等，就一个约数的对应约数存入vector中即可
        }
    sort(res.begin(), res.end());//将约数排序
    return res;
}

int main ()
{
    cin >> n;
    while (n --)
    {
        int num;
        scanf("%d", &num);
        vector<int> res = get_divisor(num);
        for (int i = 0; i < res.size(); i ++)
            printf("%d ", res[i]);
        puts("");
    }
    return 0;
}
```

### 约数的个数(数学公式)

根据数学公式约数个数 = 该数各个因数的指数+1相乘
```cpp
#include <iostream>
#include <unordered_map>
#include <algorithm>

using namespace std;

typedef long long LL;
const int mod = 1e9 + 7;

int n;

int main ()
{
    cin >> n;
    unordered_map<int, int> primes;//primes中存放因子的底数和指数
    while (n --)
    {
        int x;
        cin >> x;
        //分解质因数
        for (int i = 2; i <= x / i; i ++)
            if(x % i == 0)
                while (x % i == 0)
                {
                    x /= i;
                    primes[i] ++;
                }
        if(x > 1) primes[x] ++;
    }
    //约数个数由数学公式推导
    LL res = 1;
    for (auto prime : primes)
        res = res*(prime.second + 1) % mod;//根据公式(a1 + 1)*(a2 + 1)*.....*(ak + 1),a为因子的指数
    cout << res << endl;
    return 0;
}
```

**常用的求平方和的数学公式：（p1^0+p1^1...+p1^k :）**
==**while(最大指数 --) sum = sum * 底数 + 1**==

### 约数的和(数学公式)

```cpp
#include <iostream>
#include <unordered_map>
#include <algorithm>

using namespace std;

const int mod =1e9 + 7;
typedef long long LL;
int n;
int res = 1;

int main ()
{
    cin >> n;
    unordered_map<int, int> primes;
    //将所有的因子的质因数，分解分底数和指数的形式
    while (n --)
    {
        int num;
        cin >> num;
        for (int i = 2; i <= num / i; i ++)
            if(num % i == 0)
                while (num % i == 0)
                {
                    num /= i;
                    primes[i] ++;
                }
        if(num > 1) primes[num] ++;
    }
    //根据公式，约数之和 = (p1^0+p1^1...+p1^k)*....*(pk^1+pk^2+...pk^k)
    //p1^0+p1^1...+p1^k : while(最大指数 --) sum = sum * 底数 + 1
    for (auto prime : primes)
    {
        int num = prime.first;
        int index = prime.second;
        LL sum = 1;
        while (index --) sum = (sum * num + 1) % mod;
        //最后将所有项的sum再相乘
        res = res * sum % mod;
    }
    cout << res << endl;
    return 0;
}
```

### 求最大公约数(辗转相除法)(欧几里得算法)

==**0 % a = a**==

```cpp
#include <iostream>

using namespace std;

int n;

int main ()
{
    cin >> n;
    while (n --)
    {
        int a, b;
        scanf("%d%d", &a, &b);
        while (a % b)//辗转相除法
        {
            int c = a % b;
            a = b;
            b = c;
        }
        printf("%d\n", b);
    }
    return 0;
}
```

## 数的进制转换

```cpp
#include <iostream>
#include <vector>
#include <algorithm>
#include <string>

using namespace std;

//位数太高直接用字符串输出
int main ()
{
    int cnt;
    cin >> cnt;
    while (cnt --)
    {
        int a, b;//将a进制转换成b进制
        string num1_line, num2_line;//a进制的字符串和b进制的字符串
        vector<int> num1, num2;
        cin >> a >> b >> num1_line;
        //将num1变成数组的形式
        for (int i = 0; i < num1_line.size(); i ++)
        {
            if(num1_line[i]<= '9' && num1_line[i] >= '0') num1.push_back(num1_line[i] - '0');
            if(num1_line[i]<= 'Z' && num1_line[i] >= 'A') num1.push_back(num1_line[i] - 'A' + 10);
            if(num1_line[i]<= 'z' && num1_line[i] >= 'a') num1.push_back(num1_line[i] - 'a' + 36);
        }
        reverse(num1.begin(), num1.end());//数组要倒过来，因为最后剩余的0清除，如果不到过来就会让最低位的0清除
        
        //短除num1
        while (num1.size())
        {
            int t = 0;//剩余的余数
            for (int i = num1.size() - 1; i >= 0; i --)
            {
                num1[i] += t * a;
                t = num1[i] % b;
                num1[i] /= b;
            }
            num2.push_back(t);
            while (num1.size() && num1.back() == 0)//将高位的0清除
                num1.pop_back();
        }
        reverse(num2.begin(), num2.end());
        
        //将数组转换为字符串
        for (int i = 0; i < num2.size(); i ++)
        {
            if(num2[i] <= 9) num2_line += char(num2[i] + '0');
            if(num2[i] <= 35 && num2[i] >= 10) num2_line += char(num2[i] - 10 + 'A');
            if(num2[i] <= 61 && num2[i] >= 36) num2_line += char(num2[i] - 36 + 'a');
        }
        cout << a << ' ' << num1_line << endl;
        cout << b << ' ' << num2_line << endl;
        cout << endl;
    }
}
```
