### 高精度减法

```c++
#include <iostream>
#include <string>
#include <vector>

using namespace std;

int cmp_num(vector<int> num1, vector<int> num2)//比较两个数的大小，保证是大数减小数
{
    if(num1.size() != num2.size())
        return num1.size() >= num2.size();
    for (int i = num1.size() - 1; i >= 0; i --)
        if(num1[i] != num2[i])
            return num1[i] >= num2[i];
    return 1;
}

vector<int> Sub(vector<int> num1, vector<int> num2)
{
    vector<int> res;
    int t = 0;//是否借位
    for (int i = 0; i < num1.size(); i ++)
    {
        t = num1[i] - t;
        if(i < num2.size())
            t -= num2[i];
        res.push_back((t + 10) % 10);
        if(t >= 0)
            t = 0;
        else
            t = 1;
    }
    //消除剩下高位的多余0
    while (res.size() > 1 && res.back() == 0)
        res.pop_back();
        
    return res;
}

int main ()
{
    string num1, num2;
    cin >> num1 >> num2;
    
    //将数字逆置放在数组中
    vector<int> A, B;
    for (int i = num1.size() - 1; i >= 0; i --)
        A.push_back(num1[i] - '0');
    for (int i = num2.size() - 1; i >= 0; i --)
        B.push_back(num2[i] - '0');

    vector<int> C;//答案数组
    //分情况讨论A,B的大小
    if(cmp_num(A, B))//A数大
    {
        C = Sub(A, B);
    }
    else//B数大
    {
        C = Sub(B, A);
        cout << "-";
    }
    
    for (int i = C.size() - 1; i >= 0; i --)
        cout << C[i] ;
    return 0;
}
```

### 高精度乘法

```c++
#include <iostream>
#include <vector>
#include <string>

using namespace std;

vector<int> Mul(vector<int> &num1,int num2)
{
    vector<int> res;
    int t = 0;//进的位数
    for (int i = 0; i < num1.size() || t; i ++)//||t可以省去res.push_back(t),因为最后要是还有进位会再循环一次
    {
        if(i < num1.size())
            t += num1[i] * num2;
        res.push_back(t % 10);
        t /= 10;
    }
    //res.push_back(t);//将最后一位进位放到数组中
    while (res.size() > 1 && res.back() == 0)//将多余的0清空
        res.pop_back();
    return res;
}

//高精度乘法其中一个乘数num2的<= 10000,所以每次num2都都和num1的每一位相乘
int main ()
{
    string A, B;
    cin >> A >> B;
    //cin >> (int)B;直接读int的B，也可以
    //倒放存入数组
    vector<int> num1;
    int num2 = 0;
    for (int i = A.size() - 1; i >= 0; i --)
        num1.push_back(A[i] - '0');
    for (int i = 0; i < B.size(); i ++)
        num2 = num2 * 10 + (B[i] - '0');
        
    vector<int> res = Mul(num1, num2);
    
    for (int i = res.size() - 1; i >= 0; i --)
        cout << res[i];
    return 0;
}
```
