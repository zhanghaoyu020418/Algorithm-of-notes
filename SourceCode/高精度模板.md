    }

}

int main ()
{
    //用两个字符串来存放大数字，这样就不会溢出来
    string num1, num2;
    cin >> num1 >> num2;
    
    //用两个动态数组来倒着存放数字
    vector<int> A, B;
    for (int i = num1.size() - 1; i >= 0; i--)
        A.push_back(num1[i] - '0');
    for (int i = num2.size() - 1; i >= 0; i--)
        B.push_back(num2[i] - '0');
    
    //用一个数组来存放相加后的大数

    //倒着读取数组中的数字

    return 0;
}
