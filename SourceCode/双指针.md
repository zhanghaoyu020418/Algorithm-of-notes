# 双指针

### 判断子序列
[判断子序列](https://www.acwing.com/problem/content/2818/)

```cpp
#include <iostream>
#include <cstdio>
#include <algorithm>

using namespace std;

const int N = 100010;
int n, m;
int a[N], b[N];

int main ()
{
    cin >> n >> m;
    for (int i = 0; i < n; i ++)
        cin >> a[i];
    for (int i = 0; i < m; i ++)
        cin>> b[i];
        
    int ai = 0, bi = 0;
    while (ai < n && bi < m)
    {
        if(a[ai] == b[bi])
            ai ++;
        bi ++;
    }
    if(ai == n)
        cout << "Yes" << endl;
    else
        cout << "No" << endl;
    return 0;
}
```
