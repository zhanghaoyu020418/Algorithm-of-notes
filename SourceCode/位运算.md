# 位运算

### 二进制中1的个数
[二进制中1的个数](https://www.acwing.com/problem/content/803/)
```cpp
#include <iostream>
#include <algorithm>
#include <cstdio>

using namespace std;

const int N = 100010;
int n;
int a[N];

int main ()
{
    cin >> n;
    for (int i = 0; i < n; i ++)
        scanf("%d", &a[i]);
    
    for (int i = 0; i < n; i ++)
    {
        int num = 0;
        while (a[i])//当a[i]不为0的时候，判断最后一位是不是1，如果是num++
        {
            if(a[i] & 1)//&1 == 1 ,说明最后一位是1
                num ++;
            a[i] >>= 1;//然后右移一位
        }
        cout << num << ' ';
    }
    cout << endl;
    return 0;
}
```
